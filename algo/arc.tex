Arc $\vec{F}(u)$

\begin{equation}
	\vec{F}(-1) = \vec{A},\;\; \vec{F}(0) = \vec{B}, \;\; \vec{F}(1) = \vec{C}
\end{equation}

\begin{equation}
	\vec{F}(u) = \vec{B} + u\,\vec{S}_B + \frac{1}{2}u^2 \, \vec{\Gamma}_B + \vec{\Phi}(u)
\end{equation}

\begin{equation}
	\vec{\Phi}(0) = \vec{0}, \;\; \vec{\Phi}'(0) = \vec{0},\;\; \vec{\Phi}''(0) = \vec{0}
\end{equation}

\begin{equation}
	\vec{F}'(u) = \vec{S}_B + u\, \vec{\Gamma}_B 
\end{equation}
